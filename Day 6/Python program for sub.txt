#Develop a Python program for a small shop to process customer purchases.#
# Small Shop Purchase System

# Dictionary: product name -> price
products = {
    "Pen": 10,
    "Notebook": 50,
    "Pencil": 5
}

# Set: product categories
categories = {"Stationery"}

# List: cart items (tuples: (product, quantity))
cart = []

# Function to display products
def display_products():
    print("Available Products:")
    for product, price in products.items():
        print(product, ":", price)

# Function to add item to cart
def add_to_cart():
    try:
        name = input("Enter product name: ")
        if name not in products:
            raise NameError
        
        qty = int(input("Enter quantity: "))
        cart.append((name, qty))  # tuple used here
        print("Item added to cart successfully.")
    
    except ValueError:
        print("Invalid quantity! Please enter a number.")
    except NameError:
        print("Product not found in store.")
    except TypeError:
        print("Cart data type error.")

# Recursive function to calculate total bill
def calculate_total(cart_list):
    if len(cart_list) == 0:
        return 0
    product, qty = cart_list[0]
    return products[product] * qty + calculate_total(cart_list[1:])

# Function to view total bill
def view_bill():
    try:
        print("Items in Cart:")
        for item in cart:
            print(item[0], "x", item[1])
        
        total = calculate_total(cart)
        # Artificial check for ZeroDivisionError (as required)
        avg = total / len(cart)  # could raise ZeroDivisionError if cart empty
        
        print("Total Bill:", total)
    
    except ZeroDivisionError:
        print("Calculation error: division by zero")
    except TypeError:
        print("Cart data type error.")

# Menu-driven program
while True:
    print("\n1. Display Products")
    print("2. Add Item to Cart")
    print("3. View Total Bill")
    print("4. Exit")
    
    choice = input("Enter choice: ")
    
    if choice == '1':
        display_products()
    elif choice == '2':
        add_to_cart()
    elif choice == '3':
        view_bill()
    elif choice == '4':
        break
    else:
        print("Invalid choice!")
